1 /**
  2  * Typ obiektu reprezentującego produkt na liście zakupów.
  3  * @typedef {Object} Product
  4  * @property {number} id - Id produktu
  5  * @property {string} name - Nazwa produktu
  6  * @property {number} quantity - Liczba sztuk do zakupienia
  7  * @property {Date} dueDate - Data, do której produkt powinien być zakupiony
  8  * @property {boolean} purchased - Status informujący, czy produkt został zakupiony
  9  * @property {number} [price] - Opcjonalna cena za sztukę (dotyczy tylko zakupionych produktów)
 10  */
 11 
 12 /**
 13  * Zmienna przechowująca listę produktów.
 14  * @type {Product[]}
 15  */
 16 let shoppingList = [];
 17 
 18 /**
 19  * Funkcja generująca unikalne id dla nowego produktu.
 20  * @returns {number} - Unikalne id
 21  */
 22 function generateId() {
 23   return Math.floor(Math.random() * 1000000);
 24 }
 25 
 26 /**
 27  * Funkcja dodająca nowy produkt do listy zakupów.
 28  * @param {string} name - Nazwa produktu
 29  * @param {string} dueDateStr - Data, do której produkt powinien być zakupiony (format 'YYYY-MM-DD')
 30  * @param {number} quantity - Liczba sztuk do zakupienia
 31  */
 32 function addProduct(name, dueDateStr, quantity) {
 33   const id = generateId();
 34   const dueDate = new Date(dueDateStr);
 35   const product = {
 36     id,
 37     name,
 38     quantity,
 39     dueDate,
 40     purchased: false,
 41   };
 42   shoppingList.push(product);
 43 }
 44 
 45 /**
 46  * Funkcja usuwająca produkt z listy zakupów na podstawie id.
 47  * @param {number} id - Id produktu do usunięcia
 48  */
 49 function removeProduct(id) {
 50   shoppingList = shoppingList.filter(product => product.id !== id);
 51 }
 52 
 53 /**
 54  * Funkcja edytująca nazwę produktu.
 55  * @param {number} id - Id produktu do edycji
 56  * @param {string} newName - Nowa nazwa produktu
 57  */
 58 function editProductName(id, newName) {
 59   const product = shoppingList.find(product => product.id === id);
 60   if (product) {
 61     product.name = newName;
 62   }
 63 }
 64 
 65 /**

 67  * @param {number} id - Id produktu do edycji

 69  */
 70 function editProductStatus(id, newStatus) {
 71   const product = shoppingList.find(product => product.id === id);

 73     product.purchased = newStatus;
 74   }
 75 }

 89 /**
 90  * Funkcja edytująca datę zakupu produktu.
 91  * @param {number} id - Id produktu do edycji
 92  * @param {string} newDueDateStr - Nowa data zakupu w formacie 'YYYY-MM-DD'
 93  */
 94 function editProductDueDate(id, newDueDateStr) {
 95   const product = shoppingList.find(product => product.id === id);

 97     product.dueDate = new Date(newDueDateStr);
 98   }
 99 }
100 
101 /**
102  * Funkcja zmieniająca kolejność produktów na liście.
103  * @param {number} id - Id produktu do przesunięcia
104  * @param {string} direction - Kierunek przesunięcia ('up' lub 'down')
105  */
106 function changeProductOrder(id, direction) {
107   const index = shoppingList.findIndex(product => product.id === id);
108   if (index !== -1) {
109     if (direction === 'up' && index > 0) {
110       const temp = shoppingList[index];
111       shoppingList[index] = shoppingList[index - 1];
112       shoppingList[index - 1] = temp;
113     } else if (direction === 'down' && index < shoppingList.length - 1) {
114       const temp = shoppingList[index];
115       shoppingList[index] = shoppingList[index + 1];
116       shoppingList[index + 1] = temp;
117     }
118   }
119 }
120 
121 /**
122  * Funkcja zwracająca listę produktów, które powinny być zakupione dziś.
123  * @returns {Product[]} - Lista produktów do zakupu dzisiaj
124  */
125 function getProductsToBuyToday() {
126   const today = new Date().toDateString();
127   return shoppingList.filter(product => product.dueDate.toDateString() === today && !product.purchased);
128 }
129 
130 /**
131  * Funkcja umożliwiająca wprowadzenie ceny dla zakupionych produktów.
132  * @param {number} id - Id produktu
133  * @param {number} price - Cena za sztukę
134  */
135 function setProductPrice(id, price) {
136   const product = shoppingList.find(product => product.id === id);
137   if (product && product.purchased) {
138     product.price = price;
139   }
140 }
141 
142 /**
143  * Funkcja obliczająca koszt zakupów danego dnia.
144  * @param {string} dateStr - Data w formacie 'YYYY-MM-DD' dla której obliczany jest koszt
145  * @returns {number} - Łączny koszt zakupów
146  */
147 function calculateDailyCost(dateStr) {
148   const date = new Date(dateStr);
149   const productsForTheDay = shoppingList.filter(product => {
150     return product.dueDate.toDateString() === date.toDateString() && product.purchased;
151   });
152   let totalCost = 0;
153   productsForTheDay.forEach(product => {
154     if (product.price) {
155       totalCost += product.price * product.quantity;
156     }
157   });
158   return totalCost;
159 }
160 
161 /**
162  * Funkcja masowo modyfikująca produkty na liście na podstawie ich id.
163  * @param {number[]} ids - Lista id produktów do modyfikacji
164  * @param {function} modifyFunction - Funkcja modyfikująca pojedynczy produkt
165  */
166 function bulkModifyProducts(ids, modifyFunction) {
167   shoppingList.forEach(product => {
168     if (ids.includes(product.id)) {
169       modifyFunction(product);
170     }
171   });
172 }
